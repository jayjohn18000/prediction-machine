## Appendix: SQL used (pmci + public)

```sql
-- 1) Unique outcomes (provider_markets with snapshot in last 24h)
SELECT COUNT(DISTINCT s.provider_market_id) AS unique_outcomes_24h
FROM pmci.provider_market_snapshots s
WHERE s.observed_at > NOW() - INTERVAL '24 hours';

-- 2) Family count
SELECT COUNT(*) AS family_count FROM pmci.market_families;

-- 3) Edge window count
SELECT COUNT(*) AS edge_window_count FROM public.edge_windows;

-- 4) Overlap breakdown (by canonical_event_id)
WITH family_providers AS (
  SELECT mf.canonical_event_id, mf.id AS family_id,
    BOOL_OR(p.code = 'kalshi') AS has_kalshi,
    BOOL_OR(p.code = 'polymarket') AS has_polymarket,
    COUNT(DISTINCT p.code) AS provider_count
  FROM pmci.market_families mf
  JOIN pmci.market_links ml ON ml.family_id = mf.id
  JOIN pmci.providers p ON p.id = ml.provider_id
  WHERE ml.status <> 'removed'
  GROUP BY mf.canonical_event_id, mf.id
),
event_rollup AS (
  SELECT canonical_event_id,
    COUNT(*) FILTER (WHERE has_kalshi AND has_polymarket) AS overlapping_family_count,
    COUNT(*) FILTER (WHERE provider_count = 1) AS nonoverlapping_family_count,
    COUNT(*) AS total_family_count
  FROM family_providers
  GROUP BY canonical_event_id
)
SELECT * FROM event_rollup
ORDER BY overlapping_family_count DESC, total_family_count DESC
LIMIT 50;

-- 5) Unmatched by provider
SELECT p.code AS provider, COUNT(*) AS unmatched_count
FROM pmci.unmatched_markets um
JOIN pmci.providers p ON p.id = um.provider_id
WHERE um.active = true
GROUP BY p.code;

-- 6) Edge activity (top)
SELECT event_id, candidate, COUNT(*) AS windows,
       AVG(avg_edge) AS avg_edge, AVG(duration_seconds) AS avg_duration_s
FROM public.edge_windows
GROUP BY event_id, candidate
ORDER BY windows DESC
LIMIT 25;
```
